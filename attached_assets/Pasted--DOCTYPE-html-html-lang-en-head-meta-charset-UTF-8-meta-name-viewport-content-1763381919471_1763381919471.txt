    <script>
        // Enhanced interactive features
        const promptTextarea = document.getElementById('promptTextarea');
        const loadingDots = document.getElementById('loadingDots');
        let typingTimer;

        // Show loading indicator on typing
        promptTextarea.addEventListener('input', function() {
            clearTimeout(typingTimer);
            
            if (this.value.length > 0) {
                loadingDots.classList.add('active');
                
                typingTimer = setTimeout(() => {
                    loadingDots.classList.remove('active');
                }, 1500);
            }
        });

        // Auto-resize textarea
        promptTextarea.addEventListener('input', function() {
            this.style.height = 'auto';
            this.style.height = Math.min(this.scrollHeight, 300) + 'px';
        });

        // Button click effects
        const buttons = document.querySelectorAll('.control-button, .icon-button');
        buttons.forEach(button => {
            button.addEventListener('click', function(e) {
                // Create ripple effect
                const ripple = document.createElement('span');
                ripple.style.cssText = `
                    position: absolute;
                    background: rgba(99, 102, 241, 0.3);
                    border-radius: 50%;
                    transform: translate(-50%, -50%);
                    pointer-events: none;
                    animation: ripple 0.6s ease-out;
                `;
                
                const rect = this.getBoundingClientRect();
                const size = Math.max(rect.width, rect.height);
                ripple.style.width = ripple.style.height = size + 'px';
                ripple.style.left = (e.clientX - rect.left) + 'px';
                ripple.style.top = (e.clientY - rect.top) + 'px';
                
                this.appendChild(ripple);
                
                setTimeout(() => ripple.remove(), 600);
            });
        });

        // Add ripple animation
        const style = document.createElement('style');
        style.textContent = `
            @keyframes ripple {
                0% {
                    transform: translate(-50%, -50%) scale(0);
                    opacity: 1;
                }
                100% {
                    transform: translate(-50%, -50%) scale(4);
                    opacity: 0;
                }
            }
        `;
        document.head.appendChild(style);

        // Parallax effect on mouse move
        document.addEventListener('mousemove', (e) => {
            const x = (e.clientX / window.innerWidth - 0.5) * 20;
            const y = (e.clientY / window.innerHeight - 0.5) * 20;
            
            document.querySelector('.gradient-bg').style.transform = 
                `translate(${x}px, ${y}px) scale(1.1)`;
        });

        // Focus effect enhancement
        promptTextarea.addEventListener('focus', () => {
            document.querySelector('.main-container').style.transform = 'scale(1.02)';
        });

        promptTextarea.addEventListener('blur', () => {
            document.querySelector('.main-container').style.transform = 'scale(1)';
        });
    </script>
