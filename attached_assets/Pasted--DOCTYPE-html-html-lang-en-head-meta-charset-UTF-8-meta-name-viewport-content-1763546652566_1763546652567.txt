        /**
         * AURA QUANTUM CORE ENGINE
         * Handles Animations, Data Simulation, and Canvas Renderers
         */

        document.addEventListener('DOMContentLoaded', () => {
            // --- 1. INIT GSAP & SCROLL ---
            gsap.registerPlugin(ScrollTrigger);

            // Hero Animations
            const heroTl = gsap.timeline();
            heroTl.to('.hero-anim', {
                y: '0%',
                opacity: 1,
                duration: 1.2,
                stagger: 0.1,
                ease: 'power4.out'
            });

            // Split Text Reveal (Philosophy Section)
            const splitText = document.querySelector('.split-reveal');
            // Simple word wrap logic for animation since we don't have SplitText plugin
            const words = splitText.innerText.split(' ');
            splitText.innerHTML = '';
            words.forEach(word => {
                const span = document.createElement('span');
                span.innerText = word + ' ';
                span.style.display = 'inline-block';
                span.style.opacity = '0';
                span.style.transform = 'translateY(20px)';
                splitText.appendChild(span);
            });

            gsap.to('.split-reveal span', {
                scrollTrigger: {
                    trigger: '#about',
                    start: 'top 75%',
                },
                opacity: 1,
                y: 0,
                duration: 0.8,
                stagger: 0.02,
                ease: 'power2.out'
            });

            // Counter Animation
            gsap.utils.toArray('.counter').forEach(counter => {
                const target = parseInt(counter.getAttribute('data-target'));
                gsap.to(counter, {
                    scrollTrigger: {
                        trigger: counter,
                        start: 'top 85%',
                    },
                    innerText: target,
                    duration: 2,
                    snap: { innerText: 1 },
                    ease: 'power2.out'
                });
            });

            // Service Cards Hover Effect Logic (Additional JS for smoothness)
            const cards = document.querySelectorAll('.service-card');
            cards.forEach(card => {
                gsap.from(card, {
                    scrollTrigger: {
                        trigger: '#services',
                        start: 'top 70%'
                    },
                    y: 100,
                    opacity: 0,
                    duration: 1,
                    ease: 'power3.out',

            // --- 3. HERO CANVAS NETWORK SIMULATION ---
            const canvas = document.getElementById('network-canvas');
            const ctx = canvas.getContext('2d');
            let width, height;
            
            // Physics settings
            const PARTICLE_COUNT = 60;
            const CONNECTION_DIST = 150;
            const MOUSE_DIST = 200;
            
            let particles = [];

            class Particle {
                constructor() {
                    this.x = Math.random() * width;
                    this.y = Math.random() * height;
                    this.vx = (Math.random() - 0.5) * 0.5;
                    this.vy = (Math.random() - 0.5) * 0.5;
                    this.size = Math.random() * 2 + 1;
                }

                update() {
                    this.x += this.vx;
                    this.y += this.vy;

                    // Bounce off walls
                    if (this.x < 0 || this.x > width) this.vx *= -1;
                    if (this.y < 0 || this.y > height) this.vy *= -1;

                    // Mouse repulsion
                    let dx = mouseX - this.x;
                    let dy = mouseY - this.y;
                    let distance = Math.sqrt(dx*dx + dy*dy);

                    if (distance < MOUSE_DIST) {
                        const forceDirectionX = dx / distance;
                        const forceDirectionY = dy / distance;
                        const force = (MOUSE_DIST - distance) / MOUSE_DIST;
                        const repulsionStrength = 0.8; // Strength

                        this.vx -= forceDirectionX * force * repulsionStrength;
                        this.vy -= forceDirectionY * force * repulsionStrength;
                    }

                    // Friction to stabilize
                    this.vx *= 0.99;
                    this.vy *= 0.99;
                }

                draw() {
                    ctx.beginPath();
                    ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                    ctx.fillStyle = '#1e293b';
                    ctx.fill();
                }
            }

            function initCanvas() {
                width = window.innerWidth;
                height = window.innerHeight;
                canvas.width = width;
                canvas.height = height;
                
                particles = [];
                for(let i=0; i<PARTICLE_COUNT; i++) {
                    particles.push(new Particle());
                }
            }

            function animateCanvas() {
                ctx.clearRect(0, 0, width, height);
                
                particles.forEach(p => {
                    p.update();
                    p.draw();
                });

                // Draw connections
                for(let i=0; i<particles.length; i++) {
                    for(let j=i; j<particles.length; j++) {
                        let dx = particles[i].x - particles[j].x;
                        let dy = particles[i].y - particles[j].y;
                        let distance = Math.sqrt(dx*dx + dy*dy);

                        if (distance < CONNECTION_DIST) {
                            ctx.beginPath();
                            let opacity = 1 - (distance / CONNECTION_DIST);
                            ctx.strokeStyle = `rgba(37, 99, 235, ${opacity * 0.4})`; // Blue tint
                            ctx.lineWidth = 1;
                            ctx.moveTo(particles[i].x, particles[i].y);
                            ctx.lineTo(particles[j].x, particles[j].y);
                            ctx.stroke();
                        }
                    }
                }
                requestAnimationFrame(animateCanvas);
            }

            window.addEventListener('resize', initCanvas);
            initCanvas();
            animateCanvas();
